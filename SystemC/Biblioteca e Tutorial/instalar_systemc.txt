Para compilar:

 $ export SYSTEMC_HOME=/usr/local/systemc-2.3.1/

 $ g++ -I. -I$SYSTEMC_HOME/include -L. -L$SYSTEMC_HOME/lib-linux64 -Wl,-rpath=$SYSTEMC_HOME/lib-linux64 -o hello hello.cpp -lsystemc -lm

Para executar:
 
 $ ./hello
